const express = require("express");
const router = express.Router();
const dataService = require("./groupsService");
const data = dataService();
 
// get list of groups
router.get('/groups', (req, res, next) => {
    data.getGroups().then(data => {
        res.json(data);
    }).catch(err => {
        res.send({ 'error': err })
    })
});
 
// add group

});
